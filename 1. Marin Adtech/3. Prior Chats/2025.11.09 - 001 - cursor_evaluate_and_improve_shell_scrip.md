#!/bin/bash

# Exit on error
set -e
set -u
set -o pipefail

echo "üåÖ Good morning! Starting daily sync routine..."

# Get today's date
today=$(date +'%Y-%m-%d')
# Ensure we are on our most recent working branch ("feat-" or "BUG-")
current_branch=$(git branch --show-current)

# Validate current branch
if [ -z "$current_branch" ]; then
  echo "‚ùå Error: Not on any branch. Please checkout a branch first."
  exit 1
fi

if [ "$current_branch" = "develop" ]; then
  echo "‚ö†Ô∏è  Warning: Already on develop branch. Switching to most recent feature/bug branch..."
  # Try to find most recent feat- or BUG- branch
  recent_branch=$(git branch --list 'feat-*' 'BUG-*' --sort=-committerdate | head -n1 | sed 's/^[ *]*//')
  if [ -n "$recent_branch" ]; then
    current_branch="$recent_branch"
    git checkout "$current_branch"
    echo "‚úÖ Switched to $current_branch"
  else
    echo "‚ùå Error: No feat- or BUG- branch found. Please create one first."
    exit 1
  fi
fi

echo "üìã Current branch: $current_branch"

# Check if there are uncommitted changes
if ! git diff --quiet || ! git diff --cached --quiet; then
  echo "üì¶ Stashing local changes..."
  git stash push -m "$today-Daily-Sync-Stash"
  has_stash=true
else
  echo "‚úÖ No local changes to stash."
  has_stash=false
fi

# Fetch and update develop branch
echo "üîÑ Fetching latest changes from origin..."
git fetch origin

echo "üì• Updating develop branch..."
git checkout develop
git pull origin develop

# Merge develop into working branch
echo "üîÄ Merging develop into $current_branch..."
git checkout "$current_branch"
if ! git merge develop --no-edit; then
  echo "‚ö†Ô∏è  Merge conflicts detected! Please resolve conflicts manually."
  echo "üí° After resolving, run: git add . && git commit"
  exit 1
fi

# Restore stashed changes if any
if [ "$has_stash" = true ]; then
  echo "üì§ Restoring stashed changes..."
  if git stash list | grep -q "$today-Daily-Sync-Stash"; then
    git stash pop
    echo "‚úÖ Stashed changes restored."
  else
    echo "‚ö†Ô∏è  Warning: Expected stash not found."
  fi
fi

echo "‚úÖ Daily sync complete!"
